#include <stdio.h>

int main() {
    int inteiro = 10;
    float real = 5.5;
    char caractere = 'A';

    int *ptr_inteiro = &inteiro;
    float *ptr_real = &real;
    char *ptr_caractere = &caractere;

    // Antes da modificação
    printf("Antes:\n");
    printf("Inteiro: %d, Real: %.2f, Char: %c\n", inteiro, real, caractere);

    // Modificação usando ponteiros
    *ptr_inteiro = 20;
    *ptr_real = 10.5;
    *ptr_caractere = 'B';

    // Depois da modificação
    printf("Depois:\n");
    printf("Inteiro: %d, Real: %.2f, Char: %c\n", inteiro, real, caractere);

    return 0;
}

#include <stdio.h>

void troca(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int a = 5, b = 10;

    printf("Antes da troca: A = %d, B = %d\n", a, b);
    troca(&a, &b);
    printf("Depois da troca: A = %d, B = %d\n", a, b);

    return 0;
}

#include <stdio.h>

int string_ocorre(char *str1, char *str2) {
    while (*str1) {
        char *p1 = str1;
        char *p2 = str2;
        
        while (*p1 && *p2 && (*p1 == *p2)) {
            p1++;
            p2++;
        }

        if (!*p2) {
            return 1;  // Encontrado
        }
        str1++;
    }
    return 0;  // Não encontrado
}

int main() {
    char str1[] = "Programacao Estruturada";
    char str2[] = "Estruturada";

    if (string_ocorre(str1, str2)) {
        printf("A string \"%s\" ocorre dentro de \"%s\".\n", str2, str1);
    } else {
        printf("A string \"%s\" não ocorre dentro de \"%s\".\n", str2, str1);
    }

    return 0;
}

#include <stdio.h>

void frac(float num, int *inteiro, float *frac) {
    *inteiro = (int)num;
    *frac = num - *inteiro;
}

int main() {
    float num = 5.75;
    int parte_inteira;
    float parte_fracionaria;

    frac(num, &parte_inteira, &parte_fracionaria);

    printf("Número: %.2f\n", num);
    printf("Parte inteira: %d\n", parte_inteira);
    printf("Parte fracionária: %.2f\n", parte_fracionaria);

    return 0;
}
